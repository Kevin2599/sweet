Overview of helpful git commands

# Overview of remote URLs:
git remote show -v

# Update URL
git remote  set-url origin https://github.com/schreiberx/sweet.git


# Checkout URL
git checkout foobar

# Push to particular URL
git push foobar

git merge foobar/master
git fetch foobar
git pull foobar



#
# Merge from upstream (schreiberx/sweet.git) repository
#

# Add foobar repository with name foobar
git remote add upstream https://github.com/foobar/sweet.git

# Fetch data from upstream
git pull upstream

# merge from upstream
git merge upstream/master
